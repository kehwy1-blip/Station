```bash
# ====================== 1. Git全局身份配置（仅首次执行） ======================
# 设置全局提交用户名（GitHub账号名）
git config --global user.name "kehwy1-blip"
# 设置全局提交邮箱（与SSH密钥邮箱一致）
git config --global user.email "kehwy1@gmail.com"
# 校验全局配置是否生效
git config --global --list

# ====================== 2. SSH密钥生成与GitHub连通（仅首次执行） ======================
# 生成RSA公私钥，绑定邮箱，执行后三次回车（默认路径、无密码、确认无密码）
ssh-keygen -t rsa -C "kehwy1@gmail.com"
# 打印公钥完整内容，全选复制粘贴到GitHub网页SSH密钥设置
cat ~/.ssh/id_rsa.pub
# 测试本机SSH与GitHub服务器连通性，验证密钥有效性
ssh -T git@github.com
# 查看.ssh文件夹内密钥文件
ls ~/.ssh
# 一键打开ssh密钥文件夹窗口
explorer.exe ~/.ssh
# 查看全局git配置文件内容
cat ~/.gitconfig

# ====================== 3. 目录切换指令 ======================
# 切换到桌面目录 C:\Users\Administrator\Desktop
cd /c/Users/Administrator/Desktop
# 进入仓库文件夹Station
cd Station
# 回到用户根目录
cd ~

# ====================== 4. 仓库克隆相关 ======================
# 错误示例：用户名拼写错误，仓库找不到
git clone git@github.com:kehwy-bilt/Station.git
# 正确克隆仓库指令
git clone git@github.com:kehwy1-blip/Station.git
# 克隆到自定义文件夹（避免重名报错）
git clone git@github.com:kehwy1-blip/Station.git Station-new
# 删除旧的Station文件夹再重新克隆
rm -rf Station
git clone git@github.com:kehwy1-blip/Station.git

# ====================== 5. 本地仓库初始化（本次属于多余操作，不推荐） ======================
# 在当前目录创建空git仓库
git init

# ====================== 6. 远程同步、推送拉取核心指令 ======================
# 拉取GitHub远程最新代码同步到本地
git pull
# 将所有本地修改加入暂存区
git add .
# 提交修改到本地仓库，备注修改说明
git commit -m "修改内容描述"
# 将本地提交推送到远程main分支
git push origin main
# 仅拉取远程更新，不自动合并，查看差异
git fetch

# ====================== 7. 仓库状态查看辅助指令 ======================
# 查看本地文件修改状态
git status
# 查看当前绑定的远程仓库地址
git remote -v
# 筛选查看用户目录下git、ssh相关文件
ls -la ~ | grep -E "git|ssh"
# 查看桌面目录文件列表
ls /c/Users/Administrator/Desktop
```
